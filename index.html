1. Negativo de uma região

  • A partir da Imagem Original, deveria-se negativar a imagem partindo de um ponto P1(x1,y1) até um ponto P2(x2,y2).

Figure 1. Imagem Original

  • Dado a imagem, retornou-se ao usuário o numero de pixels para colunas e linhas, a fim de que os pontos inseridos pelo usuário estivessem dentro da matriz. Após isso, iterouse as linhas entre [x1, x2] e as colunas entre [y1, y2], subtraindo de 255 (limite máximo para a escala de cinza) o valor do ponto atual e salvando no mesmo. O resultado obtido dado P1(50,50) e P2(200,200) com esse algoritmo é a Imagem Final.

2. Trocar Regioes de uma imagem

  • A partir da Imagem Original, deveria-se particionar a imagem em quatro quadrantes e trocar aqueles da diagonal oposta.

Figure 3. Imagem Original

  • Para isso, iterou-se pela imagem seguindo y = [0 , largura) e x = [0 , altura/2), de forma que se o ponto P1(x,y) estivesse dentro do primeiro quadrante, ele fosse trocado de posição com o ponto P2(x + largura/2, y + largura/2) que está dentro do quarto quadrante. Do mesmo modo, se o ponto P1(x,y) estivesse dentro do segundo quadrante, deveria ser trocado com o ponto P2(x - largura/2, y + largura/2) dentro do terceiro quadrante. O resultado obtido com esse algoritmo é a Imagem Final.

Figure 4. Imagem Final

    std::cout << "nao abriu biel.png" << std::endl;
  }

  std::cout <<"numero de linhas "<< heigh << std::endl;
  std::cout <<"numero de colunas "<< lenght << std::endl;

  for (int i =0; i < heigh; i++){
      for (int j =0; j < lenght; j++){
          if(i<(heigh/2)&&j<(lenght/2)){
            uchar temp = image.at<uchar>(i+(lenght/2),j+(heigh/2));
            image.at<uchar>(i+(lenght/2),j+(heigh/2)) = image.at<uchar>(i,j);
            image.at<uchar>(i,j) = temp;
          }
          if(i>(heigh/2) &&j<(lenght/2)){
            uchar temp = image.at<uchar>(i-(lenght/2),j+(heigh/2));
            image.at<uchar>(i-(lenght/2),j+(heigh/2)) = image.at<uchar>(i,j);
            image.at<uchar>(i,j) = temp;
          }
      }
  }
  imwrite("bielTrocado.png", image);
}

Detecção de Bolhas

Primeriamente deveriamos remover todas as bolhas que tocassem as bordas da imagem. Para isso, a função removerBolhasBorda foi criada, de forma que ao encontrar um pixel branco, seria executada a função floodFill, tornando todos os pontos dessa bolha da cor do fundo da imagem.

Figure 5. Imagem Bolhas Inicial
removerBolhasBorda
 void removerBolhasBorda(Mat m){
  CvPoint pPrimeiro, pUltimo;
  int width = m.size().width;
  int height = m.size().height;

  //Primeira e ultima colunas
  for(int i = 0; i < height; i++){
    pPrimeiro.x = 0;
		pPrimeiro.y = i;
    floodFill(m, pPrimeiro, 0);
    pUltimo.x = width - 1;
		pUltimo.y = i;
    floodFill(m, pUltimo, 0);
  }
  //Primeira e ultima Linhas
  for(int j = 0; j < width; j++){
    pPrimeiro.x = j;
    pPrimeiro.y = height - 1;
    floodFill(m, pPrimeiro, 0);
    pUltimo.x = j;
    pUltimo.y = 0;
    floodFill(m, pUltimo, 0);
  }
}

O próximo passo é encontrar todos os buracos das bolhas. Então executou-se mais uma vez o codigo do Labeling com uma modificação para ao encontrar um buraco de bolha (cor 0), deveria-se checar o píxel anterior e caso não seja o fundo da imagem, ele deve incrementar a soma de bolhas com buracos e preencher toda a bolha com a cor branca (255).

Contar Bolhas Com Buraco
int contarBolhasComBuraco(Mat image){
  int rows = image.rows;
  int cols= image.cols;
  CvPoint p;
  int nobjects=0;
  for(int i=0; i<rows; i++){
    for(int j=0; j<cols; j++){
      if(image.at<uchar>(i,j) == 0){
        // achou um objeto
        if(image.at<uchar>(i,j-1) != 255){
          p.x=j;
          p.y=i-1;
          floodFill(image,p,255);
          nobjects++;
        }
        p.x=j;
        p.y=i;
        floodFill(image,p,255);
      }
    }
  }

  return nobjects;
}

4.2.1 Equalização de video

Para esse exercício, foi utilizada a função do openCV de Equalização de Histogramas. Dessa forma, bastou que o frame, após recebido da câmera, fosse dividido entre seus espectros de cor e cada um desse fosse equalizado. Assim, o resultado obtido é que as ocorrências das cores ficam mais bem divididas no intervalo de cores disponivies para a representação.

Codigo Final 4.2.1

#include <opencv2/opencv.hpp>

  }

void calcHist(Mat image){
  vector<Mat> planes;
  Mat histR, histG, histB;
  int nbins = 64;
  float range[] = {0, 256};
  const float *histrange = { range };
  bool uniform = true;
  bool acummulate = false;

  int histw = nbins, histh = nbins/2;
  Mat histImgR(histh, histw, CV_8UC3, Scalar(0,0,0));
  Mat histImgG(histh, histw, CV_8UC3, Scalar(0,0,0));
  Mat histImgB(histh, histw, CV_8UC3, Scalar(0,0,0));

  split(image,planes);
  calcHist(&planes[0], 1, 0, Mat(), histR, 1,
           &nbins, &histrange,
           uniform, acummulate);
  calcHist(&planes[1], 1, 0, Mat(), histG, 1,
           &nbins, &histrange,
           uniform, acummulate);
  calcHist(&planes[2], 1, 0, Mat(), histB, 1,
           &nbins, &histrange,
           uniform, acummulate);

  normalize(histR, histR, 0, histImgR.rows, NORM_MINMAX, -1, Mat());
  normalize(histG, histG, 0, histImgG.rows, NORM_MINMAX, -1, Mat());
  normalize(histB, histB, 0, histImgB.rows, NORM_MINMAX, -1, Mat());

  histImgR.setTo(Scalar(0));
  histImgG.setTo(Scalar(0));
  histImgB.setTo(Scalar(0));

  for(int i=0; i<nbins; i++){
    line(histImgR,
         Point(i, histh),
         Point(i, histh-cvRound(histR.at<float>(i))),
         Scalar(0, 0, 255), 1, 8, 0);
    line(histImgG,
         Point(i, histh),
         Point(i, histh-cvRound(histG.at<float>(i))),
         Scalar(0, 255, 0), 1, 8, 0);
    line(histImgB,
         Point(i, histh),
         Point(i, histh-cvRound(histB.at<float>(i))),
         Scalar(255, 0, 0), 1, 8, 0);
  }
  histImgR.copyTo(image(Rect(0, 0       ,nbins, histh)));
  histImgG.copyTo(image(Rect(0, histh   ,nbins, histh)));
  histImgB.copyTo(image(Rect(0, 2*histh ,nbins, histh)));
}

4.2.2 Detecção de Movimentos da Webcam

A realização da detecção de movimento é feita a partir da comparação do histograma de duas frames consecutivas, se houver mudança no histograma, houve movimentação. Para realizar a diferenciação dos histogramas foi utilizada o absoluto da diferença entre os dois histogramas e a função "threshold" do OpenCV , que caso a diferença seja maior que um limite, um valor pre-determinado é atribuido a intensidade do ponto. No caso, foi usado 255. Esta implementação para a experiência mostrou-se muito sensível a altas intensidades luminosas, tornando-se muito instável quanto a detecção de movimento.

Codigo para detecção de mudanças no histograma
bool houveMudanca(Mat histAnt, Mat histAtu, int nbins){
  int rows = histAnt.rows;
  bool mudou = false;
  float end= 0;
  Mat diff;
  if(rows>0){
    diff = abs(histAnt-histAtu);
    //aplica o limite para a diferença ser considerada como movimentação; tipo do threshold escolhiodo foi o Binario.
    threshold(diff,diff,4, 255, CV_THRESH_BINARY);
  //  cout<<diff<<endl;
    if(rows > 0 ){
      for(int i=0; i<nbins; i++){
        if(diff.at<float>(i)==255){
          mudou= true;
        }
      }
    }
  }

  return mudou;

}

// troca os quadrantes da imagem da DFT
void deslocaDFT(Mat& image ){
  Mat tmp, A, B, C, D;

  // se a imagem tiver tamanho impar, recorta a regiao para
  // evitar cÃ³pias de tamanho desigual
  image = image(Rect(0, 0, image.cols & -2, image.rows & -2));
  int cx = image.cols/2;
  int cy = image.rows/2;

  // reorganiza os quadrantes da transformada
  // A B   ->  D C
  // C D       B A
  A = image(Rect(0, 0, cx, cy));
  B = image(Rect(cx, 0, cx, cy));
  C = image(Rect(0, cy, cx, cy));
  D = image(Rect(cx, cy, cx, cy));

  // A <-> D
  A.copyTo(tmp);  D.copyTo(A);  tmp.copyTo(D);

  // C <-> B
  C.copyTo(tmp);  B.copyTo(C);  tmp.copyTo(B);
}
